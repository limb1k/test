Linux

1 -> 
ls /var/www/html/ 
# Вывожу список файлов в папке html

2 -> 
grep "404" /var/logs/access.log 
# Ищу все упоминания строки "404" в файле

3 -> 
cp 404.html 403.html
# Копирую файл 404.html в 403.html
sed -i 's/404/403/g' 403.html
# Команда sed для поиска и замены содержимого в файле, опция -i для редактирования файла, в кавычках
  использую команду s для замены подстроки 404 на 403 и командой g делаю глобальную замену в файле.

4 ->
chown :www-data index.html 
# Меняю группу файла с помощью :
chmod 640 index.html
# Изменяю права доступа числовым способом, где 6 - разрешить чтение и запись для пользователя,
  4 - разрешить чтение группе, 0 - запретить все всем остальным.

5 ->
cd /var/www/html/
# Перехожу в папку html для создания ссылки в ней

ln -s /var/logs/access.log mylink
# Использую команду ln и ее опцию -s для создания символьной ссылки на лог файл

6 -> 
ls -R /var/ 
# Командой ls и ее опциеей -R рекурсивно вывожу файлы из всех подкаталогов var

7 ->
cd /var/www/html/ 
# Перехожу в папку html

printf "<html>\n\t<head>\n\t\t<title>503</title>\n\t</head>\n</html>\n" > 503.html
# Командой printf добавляю содержимое в файл, и использую символы перевода строки \n и табуляции \t, для
  нужного отображения содержимого и оператором перенаправления > создаю файл.

expand -t 4 503.html
# Затем использую команду expand для замены символов табуляции на символы пробелов и ее опцию -t 4 для
  указани того, что символ табуляции должен заменяться на 4 символа пробела.
